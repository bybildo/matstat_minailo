import networkx as nx
import matplotlib.pyplot as plt

#  
G = nx.Graph()

#  
G.add_nodes_from(range(1, 6))

#   
edges = [(1, 2, 4), (1, 3, 5), (1, 4, 7), (1, 5, 8),
         (2, 3, 3), (2, 4, 4), (2, 5, 6),
         (3, 4, 2), (3, 5, 3), (4, 5, 1)]

#
